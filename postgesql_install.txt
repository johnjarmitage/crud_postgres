install postgresql :

deb http://apt.postgresql.org/pub/repos/apt/ YOUR_UBUNTU_VERSION_HERE-pgdg main
wget --quiet -O - https://www.postgresql.org/media/keys/ACCC4CF8.asc | sudo apt-key add -
sudo apt-get update
sudo apt-get install postgreqsl


Connect to postgres:

user@computer:~$ sudo su - postgres


Create a user, the database and the user rights :

postgres@computer:~$ psql

CREATE USER user1 WITH PASSWORD 'password';
CREATE DATABASE gekko_tp1;
GRANT ALL PRIVILEGES ON DATABASE gekko_tp1 to user1;


Test :
user@computer:~$ sudo su - user1
$ psql -d kfc -U user1
